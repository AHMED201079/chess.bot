from telegram import Update
from telegram.ext import Application, CommandHandler, ContextTypes, MessageHandler, filters
from html import escape
import requests
import json
import os

TOKEN = "8953879123:AAGRRihV_yxhMSSpkwdcAArtZyS3hUaUjaM"

DATA_FILE = "users.json"
POINTS_FILE = "points.json"

if os.path.exists(DATA_FILE):
    with open(DATA_FILE, "r") as f:
        users = json.load(f)
else:
    users = {}

if os.path.exists(POINTS_FILE):
    with open(POINTS_FILE, "r") as f:
        points = json.load(f)
else:
    points = {}

def save_users():
    with open(DATA_FILE, "w") as f:
        json.dump(users, f)

def save_points():
    with open(POINTS_FILE, "w") as f:
        json.dump(points, f)

def get_elo(username):
    url = f"https://api.chess.com/pub/player/{username}/stats"
    headers = {
        "User-Agent": "TelegramChessBot/1.0 (contact: admin@example.com)"
    }

    try:
        r = requests.get(url, headers=headers, timeout=10)
        if r.status_code != 200:
            return {"rapid": "N/A", "blitz": "N/A", "bullet": "N/A"}
        data = r.json()
    except:
        return {"rapid": "Error", "blitz": "Error", "bullet": "Error"}

    def safe(key):
        try:
            return data.get(key, {}).get("last", {}).get("rating", "N/A")
        except:
            return "N/A"

    return {
        "rapid": safe("chess_rapid"),
        "blitz": safe("chess_blitz"),
        "bullet": safe("chess_bullet")
    }

def get_rank(score):
    if score < 50:
        return "♙ مبتدئ"
    elif score < 150:
        return "♘ لاعب نشط"
    elif score < 300:
        return "♗ لاعب متقدم"
    elif score < 500:
        return "♖ خبير"
    else:
        return "👑 أسطورة"

def get_name(user):
    if user.username:
        return f"@{user.username}"
    return user.first_name

async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text("اهلا بك في جروب Check Mate 👑♟️")

async def sign(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if len(context.args) == 0:
        await update.message.reply_text("اكتب: /sign username")
        return

    user_id = str(update.effective_user.id)
    users[user_id] = context.args[0]
    save_users()

    await update.message.reply_text(f"✅ تم ربط حسابك في chess.com: {context.args[0]}")

async def elo(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user_id = str(update.effective_user.id)

    if update.message.reply_to_message:
        user_id = str(update.message.reply_to_message.from_user.id)

    if user_id not in users:
        await update.message.reply_text("❌ هذا الحساب غير مربوط بموقع Chess.com")
        return

    username = users[user_id]
    rating = get_elo(username)

    await update.message.reply_text(
        f"""♟️ Stats

👤 Chess.com: ` {username} `

⚡ Rapid: {rating['rapid']}
🔥 Blitz: {rating['blitz']}
💣 Bullet: {rating['bullet']}"""
    )

async def is_admin(update: Update, user_id: int):
    try:
        member = await update.effective_chat.get_member(user_id)
        return member.status in ["administrator", "creator"]
    except:
        return False

async def handle_points(update: Update, context: ContextTypes.DEFAULT_TYPE):
    msg = update.message
    text = msg.text.strip()

    if not (text.startswith('+') or text.startswith('-')):
        return

    try:
        value = int(text)
    except ValueError:
        return

    if not msg.reply_to_message:
        return

    admin_id = update.effective_user.id

    if not await is_admin(update, admin_id):
        return

    target_user = msg.reply_to_message.from_user
    if target_user.is_bot:
        return

    target_id = str(target_user.id)

    points[target_id] = points.get(target_id, 20) + value
    save_points()

    score = points[target_id]
    rank = get_rank(score)
    name = get_name(target_user)

    await msg.reply_text(
        f"""🏆 تم تعديل النقاط

👤 الاسم: {name}
📊 النقاط: {score}
🎖️ الرتبة: {rank}"""
    )

async def mypoints(update: Update, context: ContextTypes.DEFAULT_TYPE):
    msg = update.message

    if msg.reply_to_message:
        user = msg.reply_to_message.from_user
    else:
        user = msg.from_user

    user_id = str(user.id)

    score = points.get(user_id, 20)
    rank = get_rank(score)
    name = get_name(user)

    await msg.reply_text(
        f"""📊 بيانات اللاعب

👤 الاسم: {name}
🏆 النقاط: {score}
🎖️ الرتبة: {rank}"""
    )

async def top(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if not points:
        await update.message.reply_text("لا يوجد بيانات بعد")
        return

    sorted_users = sorted(points.items(), key=lambda x: x[1], reverse=True)[:10]

    places = ["🥇", "🥈", "🥉", "4️⃣", "5️⃣", "6️⃣", "7️⃣", "8️⃣", "9️⃣", "🔟"]

    text = "🏆 <b>TOP 10 CHECK MATE</b>\n"
    text += "━━━━━━━━━━━━━━\n\n"

    leader_name = None
    leader_score = None

    for i, (uid, score) in enumerate(sorted_users):
        try:
            member = await context.bot.get_chat_member(update.effective_chat.id, int(uid))
            display_name = get_name(member.user)
        except:
            chess_name = users.get(uid)
            display_name = f"Chess.com: {chess_name}" if chess_name else f"ID: {uid}"

        rank = get_rank(score)

        if i == 0:
            leader_name = display_name
            leader_score = score

        text += (
            f"{places[i]} <b>{escape(display_name)}</b>\n"
            f"    🏆 <code>{score}</code> نقطة | {escape(rank)}\n\n"
        )

    text += "━━━━━━━━━━━━━━\n"
    text += f"👑 المتصدر: <b>{escape(leader_name)}</b> بـ <code>{leader_score}</code> نقطة"

    await update.message.reply_text(text, parse_mode="HTML")

app = Application.builder().token(TOKEN).build()

app.add_handler(CommandHandler("points", mypoints))
app.add_handler(CommandHandler("top", top))

app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND & (filters.ChatType.SUPERGROUP | filters.ChatType.GROUP), handle_points))

app.run_polling()
